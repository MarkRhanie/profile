import 'package:flutter/material.dart';
import 'package:shared_preferences/shared_preferences.dart';
import 'dart:convert';

class AuthService extends ChangeNotifier {
  static const String _usersKey = 'registered_users';
  static const String _currentUserKey = 'current_user';

  Map<String, dynamic>? _currentUser;
  bool _isLoading = false;

  Map<String, dynamic>? get currentUser => _currentUser;
  bool get isLoading => _isLoading;
  bool get isLoggedIn => _currentUser != null;

  Future<void> loadSavedSession() async {
    _isLoading = true;
    notifyListeners();

    try {
      final prefs = await SharedPreferences.getInstance();
      final userJson = prefs.getString(_currentUserKey);

      if (userJson != null) {
        _currentUser = json.decode(userJson);
        print('Loaded user: $_currentUser');
      }
    } catch (e) {
      print('Error loading session: $e');
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }

  Future<List<Map<String, dynamic>>> getAllUsers() async {
    final prefs = await SharedPreferences.getInstance();
    final usersJson = prefs.getString(_usersKey);

    if (usersJson != null) {
      return List<Map<String, dynamic>>.from(json.decode(usersJson));
    }
    return [];
  }

  Future<void> _saveUsers(List<Map<String, dynamic>> users) async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_usersKey, json.encode(users));
  }

  Future<String?> register({
    required String username,
    required String email,
    required String password,
    String? fullName,
  }) async {
    try {
      final users = await getAllUsers();
      print('Existing users: $users');

      if (users.any((user) => user['email'] == email)) {
        return 'Email already exists';
      }

      if (users.any((user) => user['username'] == username)) {
        return 'Username already taken';
      }

      String defaultFullName = fullName ?? username.replaceFirst('@', '');

      final newUser = {
        'id': DateTime.now().millisecondsSinceEpoch,
        'username': username,
        'email': email,
        'password': password,
        'fullName': defaultFullName,
        'bio': '',
        'website': '',
        'location': '',
        'createdAt': DateTime.now().toIso8601String(),
      };

      users.add(newUser);
      await _saveUsers(users);
      print('New user registered: $newUser');

      return null;
    } catch (e) {
      print('Registration error: $e');
      return 'An error occurred: $e';
    }
  }

  Future<String?> login(String email, String password) async {
    try {
      final users = await getAllUsers();
      final user = users.firstWhere(
        (u) => u['email'] == email && u['password'] == password,
        orElse: () => {},
      );

      if (user.isNotEmpty) {
        print('Login successful for user: $user');
        _currentUser = user;
        await _saveCurrentSession(user);
        notifyListeners();
        return null;
      }
      return 'Invalid email or password';
    } catch (e) {
      print('Login error: $e');
      return 'An error occurred: $e';
    }
  }

  Future<String?> loginWithUsername(String username, String password) async {
    try {
      final users = await getAllUsers();
      print('Attempting login with username: $username');

      final user = users.firstWhere(
        (u) => u['username'] == username && u['password'] == password,
        orElse: () => {},
      );

      if (user.isNotEmpty) {
        print('Login successful for user: $user');
        _currentUser = user;
        await _saveCurrentSession(user);
        notifyListeners();
        return null;
      }

      return 'Invalid username or password';
    } catch (e) {
      print('Login error: $e');
      return 'An error occurred: $e';
    }
  }

  Future<void> updateProfile({
    required String displayName,
    required String bio,
    required String website,
    required String location,
    String? profileImagePath,
    String? profileImageBase64,
  }) async {
    try {
      if (_currentUser == null) return;

      final users = await getAllUsers();

      final userIndex = users.indexWhere((u) => u['id'] == _currentUser!['id']);
      if (userIndex != -1) {
        users[userIndex]['fullName'] = displayName;
        users[userIndex]['bio'] = bio;
        users[userIndex]['website'] = website;
        users[userIndex]['location'] = location;

        if (profileImagePath != null) {
          users[userIndex]['profileImagePath'] = profileImagePath;
        }

        if (profileImageBase64 != null) {
          users[userIndex]['profileImageBase64'] = profileImageBase64;
        }

        await _saveUsers(users);

        _currentUser = users[userIndex];
        await _saveCurrentSession(_currentUser!);

        notifyListeners();
      }
    } catch (e) {
      print('Error updating profile: $e');
    }
  }

  Future<void> _saveCurrentSession(Map<String, dynamic> user) async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.setString(_currentUserKey, json.encode(user));
  }

  Future<void> logout() async {
    _isLoading = true;
    notifyListeners();

    try {
      final prefs = await SharedPreferences.getInstance();
      await prefs.remove(_currentUserKey);
      _currentUser = null;
      print('User logged out');
    } catch (e) {
      print('Error logging out: $e');
    } finally {
      _isLoading = false;
      notifyListeners();
    }
  }

  Future<void> clearAllUsers() async {
    final prefs = await SharedPreferences.getInstance();
    await prefs.remove(_usersKey);
    await prefs.remove(_currentUserKey);
    _currentUser = null;
    print('All users cleared');
  }
}

    _bioController.text = user?['bio'] ?? '';

  Future<void> _loadSavedImage() async {
    final authService = Provider.of<AuthService>(context, listen: false);
    final user = authService.currentUser;

    if (kIsWeb) {
      final imageBase64 = user?['profileImageBase64'];
      if (imageBase64 != null && imageBase64.isNotEmpty) {
        try {
          final bytes = base64.decode(imageBase64);
          setState(() {
            _profileImageBytes = bytes;
            _profileImageBase64 = imageBase64;
          });
        } catch (e) {
          print('Error decoding image: $e');
        }
      }
    } else {
      final imagePath = user?['profileImagePath'];
      if (imagePath != null && imagePath.isNotEmpty) {
        try {
          final file = File(imagePath);
          if (await file.exists()) {
            setState(() {
              _profileImageFile = file;
            });
          }
        } catch (e) {
          print('Error loading file: $e');
        }
      }
    }
  }

    if (kIsWeb) return true;

  Future<void> _pickImageFromGallery() async {
    if (!kIsWeb && Theme.of(context).platform == TargetPlatform.android) {
      bool hasPermission = await _requestPermission(Permission.photos);
      if (!hasPermission) {
        _showSnackBar('Storage permission denied', isError: true);

            }
            return null;
          },

  String? _getAlbumArt(PostModel song, BuildContext context) {
    final postProvider = Provider.of<PostProvider>(context, listen: false);
    final allMusic = postProvider.getAllLocalMusic();

    for (var music in allMusic) {
      if (music['title'] == song.songTitle && music['artist'] == song.artist) {
        if (music['coverType'] == 'image' && music['coverArt'] != null) {
          return music['coverArt'];
        }
      }
    }
    return null;
  }

  String _getTimeAgo(DateTime timestamp) {
    final now = DateTime.now();
    final difference = now.difference(timestamp);
    if (difference.inDays > 0) return '${difference.inDays}d';
    if (difference.inHours > 0) return '${difference.inHours}h';
    if (difference.inMinutes > 0) return '${difference.inMinutes}m';
    return 'now';
  }

  Future<void> _handleLogout() async {
    final authService = Provider.of<AuthService>(context, listen: false);

    final bool? confirm = await showDialog(
      context: context,
      builder: (context) => AlertDialog(
        backgroundColor: AppColors.card,
        title: const Text(
          'Logout',
          style: TextStyle(color: Colors.white, fontWeight: FontWeight.bold),
        ),
        content: const Text(
          'Are you sure you want to logout?',
          style: TextStyle(color: Colors.grey, fontSize: 16),
        ),
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(20)),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(context, false),
            child: Text(
              'Cancel',
              style: TextStyle(color: Colors.grey[400], fontSize: 16),
            ),
          ),
          ElevatedButton(
            onPressed: () => Navigator.pop(context, true),
            style: ElevatedButton.styleFrom(
              backgroundColor: Colors.red,
              shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(30)),
              padding: const EdgeInsets.symmetric(horizontal: 24, vertical: 12),
            ),
            child: const Text('Logout', style: TextStyle(fontSize: 16)),
          ),
        ],
      ),
    );

      await authService.logout();
      if (mounted) {
        Navigator.of(context).pushAndRemoveUntil(
          MaterialPageRoute(builder: (_) => const LoginScreen()),
          (route) => false,
        );
      }
    }
  }

  Future<void> _showFollowersList() async {
    final followService = Provider.of<FollowService>(context, listen: false);
    final authService = Provider.of<AuthService>(context, listen: false);

    final List<Map<String, dynamic>> followers = [];
    for (String userId in followService.followersList) {
      final details = await followService.getUserDetails(userId);
      followers.add(details);
    }

    if (!mounted) return;
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (context) => FollowListScreen(
          title: 'Followers',
          users: followers,
          onFollowTap: (user) async {
            if (user['isFollowing']) {
              await followService.followUser(user['id'].toString());
            } else {
              await followService.unfollowUser(user['id'].toString());
            }
          },
        ),
      ),
    );
  }

      builder: (context, authService, followService, child) {
        final user = authService.currentUser;
        final String rawUsername = user?['username'] ?? 'spotifyuser';
        final String displayUsername = rawUsername.startsWith('@')
            ? rawUsername
            : '@$rawUsername';
        final displayName = user?['fullName'] ?? rawUsername.replaceFirst('@', '');
        final bio = user?['bio'] ?? '';

        final profileImagePath = user?['profileImagePath'];
        final profileImageBase64 = user?['profileImageBase64'];

        ImageProvider? profileImage;
        if (profileImageBase64 != null && profileImageBase64.isNotEmpty) {
          try {
            final bytes = base64.decode(profileImageBase64);
            profileImage = MemoryImage(bytes);
          } catch (e) {
            debugPrint('Error decoding profile image: $e');
          }
        } else if (profileImagePath != null && profileImagePath.isNotEmpty) {
          try {
            final file = File(profileImagePath);
            if (file.existsSync()) {
              profileImage = FileImage(file);
            }
          } catch (e) {
            debugPrint('Error loading profile image: $e');
          }
        }

        return Consumer2<PostProvider, ThreadProvider>(
          builder: (context, postProvider, threadProvider, child) {
            final userMusicPosts = postProvider.posts.where((post) =>
              post.username == rawUsername ||
              post.username == '@$rawUsername').toList();

            final userThreadPosts = threadProvider.threadPosts.where((thread) =>
              (thread.username == rawUsername || thread.username == '@$rawUsername') && 
              !thread.isQuote).toList();

            List<dynamic> allUserPosts = [];
            allUserPosts.addAll(userMusicPosts);
            allUserPosts.addAll(userThreadPosts);
            allUserPosts.sort((a, b) => b.timestamp.compareTo(a.timestamp));

            List<dynamic> allLikedPosts = [];
            allLikedPosts.addAll(postProvider.userLikedPosts);
            allLikedPosts.addAll(threadProvider.userLikedThreads);
            allLikedPosts.sort((a, b) => b.timestamp.compareTo(a.timestamp));

            allRepostedPosts.addAll(postProvider.userReposts);

  @override
  double get maxExtent => _tabBar.preferredSize.height;

  }
